104 private void ProcessAck(NetPacket packet)
105 {
106 if (packet.Size != _outgoingAcks.Size)
107 {
108 NetDebug.Write("[PA]Invalid acks packet size");
109 return;
110 }
111
112 ushort ackWindowStart = packet.Sequence;
113 int windowRel = NetUtils.RelativeSequenceNumber(_localWindowStart, ackWindowStart);
114 if (ackWindowStart >= NetConstants.MaxSequence || windowRel < 0)
115 {
116 NetDebug.Write("[PA]Bad window start");
117 return;
118 }
119
120 //check relevance
121 if (windowRel >= _windowSize)
122 {
123 NetDebug.Write("[PA]Old acks");
124 return;
125 }
126
127 byte[] acksData = packet.RawData;
128 lock (_pendingPackets)
129 {
130 for (int pendingSeq = _localWindowStart;
131 pendingSeq != _localSeqence;
132 pendingSeq = (pendingSeq + 1) % NetConstants.MaxSequence)
133 {
134 int rel = NetUtils.RelativeSequenceNumber(pendingSeq, ackWindowStart);
135 if (rel >= _windowSize)
136 {
137 NetDebug.Write("[PA]REL: " + rel);
138 break;
139 }
140
141 int pendingIdx = pendingSeq % _windowSize;
142 int currentByte = NetConstants.ChanneledHeaderSize + pendingIdx / BitsInByte;
143 int currentBit = pendingIdx % BitsInByte;
144 if ((acksData[currentByte] & (1 << currentBit)) == 0)
145 {
146 if (Peer.NetManager.EnableStatistics)
147 {
148 Peer.Statistics.IncrementPacketLoss();
149 Peer.NetManager.Statistics.IncrementPacketLoss();
150 }
151
152 //Skip false ack
153 NetDebug.Write("[PA]False ack: {0}", pendingSeq);
154 continue;
155 }
156
157 if (pendingSeq == _localWindowStart)
158 {
159 //Move window
160 _localWindowStart = (_localWindowStart + 1) % NetConstants.MaxSequence;
161 }
162
163 //clear packet
164 if (_pendingPackets[pendingIdx].Clear(Peer))
165 NetDebug.Write("[PA]Removing reliableInOrder ack: {0} - true", pendingSeq);
166 }
167 }

170 protected override bool SendNextPackets()
171 {
172 if (_mustSendAcks)
173 {
174 _mustSendAcks = false;
175 NetDebug.Write("[RR]SendAcks");
176 lock(_outgoingAcks)
177 Peer.SendUserData(_outgoingAcks);

179
180 long currentTime = DateTime.UtcNow.Ticks;
181 bool hasPendingPackets = false;
182
183 lock (_pendingPackets)
184 {
185 //get packets from queue
186 lock (OutgoingQueue)
187 {
188 while (OutgoingQueue.Count > 0)
189 {
190 int relate = NetUtils.RelativeSequenceNumber(_localSeqence, _localWindowStart);
191 if (relate >= _windowSize)
192 break;
193
194 var netPacket = OutgoingQueue.Dequeue();
195 netPacket.Sequence = (ushort) _localSeqence;
196 netPacket.ChannelId = _id;
197 _pendingPackets[_localSeqence % _windowSize].Init(netPacket);
198 _localSeqence = (_localSeqence + 1) % NetConstants.MaxSequence;
199 }
200 }
201
202 //send
203 for (int pendingSeq = _localWindowStart; pendingSeq != _localSeqence; pendingSeq = (pendingSeq + 1) % NetConstants.MaxSequence)
204 {
205 // Please note: TrySend is invoked on a mutable struct, it's important to not extract it into a variable here
206 bool hasPacket;
207 _pendingPackets[pendingSeq % _windowSize].TrySend(currentTime, Peer, out hasPacket);
208 if (hasPacket)
209 {
210 hasPendingPackets = true;
211 }
212 }
213 }
214

219 public override bool ProcessPacket(NetPacket packet)
220 {
221 if (packet.Property == PacketProperty.Ack)
222 {
223 ProcessAck(packet);
224 return false;
225 }
226 int seq = packet.Sequence;
227 if (seq >= NetConstants.MaxSequence)
228 {
229 NetDebug.Write("[RR]Bad sequence");
230 return false;
231 }
232
233 int relate = NetUtils.RelativeSequenceNumber(seq, _remoteWindowStart);
234 int relateSeq = NetUtils.RelativeSequenceNumber(seq, _remoteSequence);
235
236 if (relateSeq > _windowSize)
237 {
238 NetDebug.Write("[RR]Bad sequence");
239 return false;
240 }
241
242 //Drop bad packets
243 if (relate < 0)
244 {
245 //Too old packet doesn't ack
246 NetDebug.Write("[RR]ReliableInOrder too old");
247 return false;
248 }
249 if (relate >= _windowSize * 2)
250 {
251 //Some very new packet
252 NetDebug.Write("[RR]ReliableInOrder too new");
253 return false;
254 }
255
256 //If very new - move window
257 int ackIdx;
258 int ackByte;
259 int ackBit;
260 lock (_outgoingAcks)
261 {
262 if (relate >= _windowSize)
263 {
264 //New window position
265 int newWindowStart = (_remoteWindowStart + relate - _windowSize + 1) % NetConstants.MaxSequence;
266 _outgoingAcks.Sequence = (ushort) newWindowStart;
267
268 //Clean old data
269 while (_remoteWindowStart != newWindowStart)
270 {
271 ackIdx = _remoteWindowStart % _windowSize;
272 ackByte = NetConstants.ChanneledHeaderSize + ackIdx / BitsInByte;
273 ackBit = ackIdx % BitsInByte;
274 _outgoingAcks.RawData[ackByte] &= (byte) ~(1 << ackBit);
275 _remoteWindowStart = (_remoteWindowStart + 1) % NetConstants.MaxSequence;
276 }
277 }
278
279 //Final stage - process valid packet
280 //trigger acks send
281 _mustSendAcks = true;
282
283 ackIdx = seq % _windowSize;
284 ackByte = NetConstants.ChanneledHeaderSize + ackIdx / BitsInByte;
285 ackBit = ackIdx % BitsInByte;
286 if ((_outgoingAcks.RawData[ackByte] & (1 << ackBit)) != 0)
287 {
288 NetDebug.Write("[RR]ReliableInOrder duplicate");
289 return false;
290 }
291
292 //save ack
293 _outgoingAcks.RawData[ackByte] |= (byte) (1 << ackBit);
294 }
295
296 AddToPeerChannelSendQueue();
297
298 //detailed check
299 if (seq == _remoteSequence)
300 {
301 NetDebug.Write("[RR]ReliableInOrder packet succes");
302 Peer.AddReliablePacket(_deliveryMethod, packet);
303 _remoteSequence = (_remoteSequence + 1) % NetConstants.MaxSequence;
304
305 if (_ordered)
306 {
307 NetPacket p;
308 while ((p = _receivedPackets[_remoteSequence % _windowSize]) != null)
309 {
310 //process holden packet
311 _receivedPackets[_remoteSequence % _windowSize] = null;
312 Peer.AddReliablePacket(_deliveryMethod, p);
313 _remoteSequence = (_remoteSequence + 1) % NetConstants.MaxSequence;
314 }
315 }
316 else
317 {
318 while (_earlyReceived[_remoteSequence % _windowSize])
319 {
320 //process early packet
321 _earlyReceived[_remoteSequence % _windowSize] = false;
322 _remoteSequence = (_remoteSequence + 1) % NetConstants.MaxSequence;
323 }
324 }
325 return true;
326 }
327
328 //holden packet
329 if (_ordered)
330 {
331 _receivedPackets[ackIdx] = packet;
332 }
333 else
334 {
335 _earlyReceived[ackIdx] = true;
336 Peer.AddReliablePacket(_deliveryMethod, packet);
337 }
338 return true;
339 }
