253 private BaseChannel CreateChannel(byte idx)
254 {
255 BaseChannel newChannel = _channels[idx];
256 if (newChannel != null)
257 return newChannel;
258 switch ((DeliveryMethod)(idx % 4))
259 {
260 case DeliveryMethod.ReliableUnordered:
261 newChannel = new ReliableChannel(this, false, idx);
262 break;
263 case DeliveryMethod.Sequenced:
264 newChannel = new SequencedChannel(this, false, idx);
265 break;
266 case DeliveryMethod.ReliableOrdered:
267 newChannel = new ReliableChannel(this, true, idx);
268 break;
269 case DeliveryMethod.ReliableSequenced:
270 newChannel = new SequencedChannel(this, true, idx);
271 break;
272 }
273 BaseChannel prevChannel = Interlocked.CompareExchange(ref _channels[idx], newChannel, null);
274 if (prevChannel != null)
275 return prevChannel;
276
277 return newChannel;
278 }

502 private void SendInternal(
503 byte[] data,
504 int start,
505 int length,
506 byte channelNumber,
507 DeliveryMethod deliveryMethod,
508 object userData)
509 {
510 if (_connectionState != ConnectionState.Connected || channelNumber >= _channels.Length)
511 return;
512
513 //Select channel
514 PacketProperty property;
515 BaseChannel channel = null;
516
517 if (deliveryMethod == DeliveryMethod.Unreliable)
518 {
519 property = PacketProperty.Unreliable;
520 }
521 else
522 {
523 property = PacketProperty.Channeled;
524 channel = CreateChannel((byte)(channelNumber*4 + (byte)deliveryMethod));
525 }
526
527 //Prepare
528 NetDebug.Write("[RS]Packet: " + property);
529
530 //Check fragmentation
531 int headerSize = NetPacket.GetHeaderSize(property);
532 //Save mtu for multithread
533 int mtu = _mtu;
534 if (length + headerSize > mtu)
535 {
536 //if cannot be fragmented
537 if (deliveryMethod != DeliveryMethod.ReliableOrdered && deliveryMethod != DeliveryMethod.ReliableUnordered)
538 throw new TooBigPacketException("Unreliable or ReliableSequenced packet size exceeded maximum of " + (mtu - headerSize) + " bytes, Check allowed size by GetMaxSinglePacketSize()");
539
540 int packetFullSize = mtu - headerSize;
541 int packetDataSize = packetFullSize - NetConstants.FragmentHeaderSize;
542 int totalPackets = length / packetDataSize + (length % packetDataSize == 0 ? 0 : 1);
543
544 NetDebug.Write("FragmentSend:\n" +
545 " MTU: {0}\n" +
546 " headerSize: {1}\n" +
547 " packetFullSize: {2}\n" +
548 " packetDataSize: {3}\n" +
549 " totalPackets: {4}",
550 mtu, headerSize, packetFullSize, packetDataSize, totalPackets);
551
552 if (totalPackets > ushort.MaxValue)
553 throw new TooBigPacketException("Data was split in " + totalPackets + " fragments, which exceeds " + ushort.MaxValue);
554
555 ushort currentFragmentId = (ushort)Interlocked.Increment(ref _fragmentId);
556
557 for(ushort partIdx = 0; partIdx < totalPackets; partIdx++)
558 {
559 int sendLength = length > packetDataSize ? packetDataSize : length;
560
561 NetPacket p = _packetPool.GetPacket(headerSize + sendLength + NetConstants.FragmentHeaderSize);
562 p.Property = property;
563 p.UserData = userData;
564 p.FragmentId = currentFragmentId;
565 p.FragmentPart = partIdx;
566 p.FragmentsTotal = (ushort)totalPackets;
567 p.MarkFragmented();
568
569 Buffer.BlockCopy(data, start + partIdx * packetDataSize, p.RawData, NetConstants.FragmentedHeaderTotalSize, sendLength);
570 channel.AddToQueue(p);
571
572 length -= sendLength;
573 }
574 return;
575 }
576
577 //Else just send
578 NetPacket packet = _packetPool.GetPacket(headerSize + length);
579 packet.Property = property;
580 Buffer.BlockCopy(data, start, packet.RawData, headerSize, length);
581 packet.UserData = userData;
582
583 if (channel == null) //unreliable
584 {
585 lock(_unreliableChannel)
586 _unreliableChannel.Enqueue(packet);
587 }
588 else
589 {
590 channel.AddToQueue(packet);
591 }
592 }

636 internal ShutdownResult Shutdown(byte[] data, int start, int length, bool force)
637 {
638 lock (_shutdownLock)
639 {
640 //trying to shutdown already disconnected
641 if (_connectionState == ConnectionState.Disconnected ||
642 _connectionState == ConnectionState.ShutdownRequested)
643 {
644 return ShutdownResult.None;
645 }
646
647 var result = _connectionState == ConnectionState.Connected
648 ? ShutdownResult.WasConnected
649 : ShutdownResult.Success;
650
651 //don't send anything
652 if (force)
653 {
654 _connectionState = ConnectionState.Disconnected;
655 return result;
656 }
657
658 //reset time for reconnect protection
659 Interlocked.Exchange(ref _timeSinceLastPacket, 0);
660
661 //send shutdown packet
662 _shutdownPacket = new NetPacket(PacketProperty.Disconnect, length) {ConnectionNumber = _connectNum};
663 FastBitConverter.GetBytes(_shutdownPacket.RawData, 1, _connectTime);
664 if (_shutdownPacket.Size >= _mtu)
665 {
666 //Drop additional data
667 NetDebug.WriteError("[Peer] Disconnect additional data size more than MTU - 8!");
668 }
669 else if (data != null && length > 0)
670 {
671 Buffer.BlockCopy(data, start, _shutdownPacket.RawData, 9, length);
672 }
673 _connectionState = ConnectionState.ShutdownRequested;
674 NetDebug.Write("[Peer] Send disconnect");
675 NetManager.SendRaw(_shutdownPacket, EndPoint);
676 return result;
677 }

688 internal void AddReliablePacket(DeliveryMethod method, NetPacket p)
689 {
690 if (p.IsFragmented)
691 {
692 NetDebug.Write("Fragment. Id: {0}, Part: {1}, Total: {2}", p.FragmentId, p.FragmentPart, p.FragmentsTotal);
693 //Get needed array from dictionary
694 ushort packetFragId = p.FragmentId;
695 IncomingFragments incomingFragments;
696 if (!_holdedFragments.TryGetValue(packetFragId, out incomingFragments))
697 {
698 incomingFragments = new IncomingFragments
699 {
700 Fragments = new NetPacket[p.FragmentsTotal],
701 ChannelId = p.ChannelId
702 };
703 _holdedFragments.Add(packetFragId, incomingFragments);
704 }
705
706 //Cache
707 var fragments = incomingFragments.Fragments;
708
709 //Error check
710 if (p.FragmentPart >= fragments.Length ||
711 fragments[p.FragmentPart] != null ||
712 p.ChannelId != incomingFragments.ChannelId)
713 {
714 _packetPool.Recycle(p);
715 NetDebug.WriteError("Invalid fragment packet");
716 return;
717 }
718 //Fill array
719 fragments[p.FragmentPart] = p;
720
721 //Increase received fragments count
722 incomingFragments.ReceivedCount++;
723
724 //Increase total size
725 incomingFragments.TotalSize += p.Size - NetConstants.FragmentedHeaderTotalSize;
726
727 //Check for finish
728 if (incomingFragments.ReceivedCount != fragments.Length)
729 return;
730
731 //just simple packet
732 NetPacket resultingPacket = _packetPool.GetPacket(incomingFragments.TotalSize);
733
734 int pos = 0;
735 for (int i = 0; i < incomingFragments.ReceivedCount; i++)
736 {
737 var fragment = fragments[i];
738 int writtenSize = fragment.Size - NetConstants.FragmentedHeaderTotalSize;
739
740 if (pos+writtenSize > resultingPacket.RawData.Length)
741 {
742 _holdedFragments.Remove(packetFragId);
743 NetDebug.WriteError("Fragment error pos: {0} >= resultPacketSize: {1} , totalSize: {2}",
744 pos + writtenSize,
745 resultingPacket.RawData.Length,
746 incomingFragments.TotalSize);
747 return;
748 }
749
750 //Create resulting big packet
751 Buffer.BlockCopy(
752 fragment.RawData,
753 NetConstants.FragmentedHeaderTotalSize,
754 resultingPacket.RawData,
755 pos,
756 writtenSize);
757 pos += writtenSize;
758
759 //Free memory
760 _packetPool.Recycle(fragment);
761 fragments[i] = null;
762 }
763
764 //Clear memory
765 _holdedFragments.Remove(packetFragId);
766
767 //Send to process
768 NetManager.CreateReceiveEvent(resultingPacket, method, 0, this);
769 }
770 else //Just simple packet
771 {
772 NetManager.CreateReceiveEvent(p, method, NetConstants.ChanneledHeaderSize, this);
773 }
774 }

776 private void ProcessMtuPacket(NetPacket packet)
777 {
778 //header + int
779 if (packet.Size < NetConstants.PossibleMtu[0])
780 return;
781
782 //first stage check (mtu check and mtu ok)
783 int receivedMtu = BitConverter.ToInt32(packet.RawData, 1);
784 int endMtuCheck = BitConverter.ToInt32(packet.RawData, packet.Size - 4);
785 if (receivedMtu != packet.Size || receivedMtu != endMtuCheck || receivedMtu > NetConstants.MaxPacketSize)
786 {
787 NetDebug.WriteError("[MTU] Broken packet. RMTU {0}, EMTU {1}, PSIZE {2}", receivedMtu, endMtuCheck, packet.Size);
788 return;
789 }
790
791 if (packet.Property == PacketProperty.MtuCheck)
792 {
793 _mtuCheckAttempts = 0;
794 NetDebug.Write("[MTU] check. send back: " + receivedMtu);
795 packet.Property = PacketProperty.MtuOk;
796 NetManager.SendRawAndRecycle(packet, EndPoint);
797 }
798 else if(receivedMtu > _mtu && !_finishMtu) //MtuOk
799 {
800 //invalid packet
801 if (receivedMtu != NetConstants.PossibleMtu[_mtuIdx + 1])
802 return;
803
804 lock (_mtuMutex)
805 {
806 SetMtu(_mtuIdx+1);
807 }
808 //if maxed - finish.
809 if (_mtuIdx == NetConstants.PossibleMtu.Length - 1)
810 _finishMtu = true;
811
812 NetDebug.Write("[MTU] ok. Increase to: " + _mtu);
813 }
814 }

816 private void UpdateMtuLogic(int deltaTime)
817 {
818 if (_finishMtu)
819 return;
820
821 _mtuCheckTimer += deltaTime;
822 if (_mtuCheckTimer < MtuCheckDelay)
823 return;
824
825 _mtuCheckTimer = 0;
826 _mtuCheckAttempts++;
827 if (_mtuCheckAttempts >= MaxMtuCheckAttempts)
828 {
829 _finishMtu = true;
830 return;
831 }
832
833 lock (_mtuMutex)
834 {
835 if (_mtuIdx >= NetConstants.PossibleMtu.Length - 1)
836 return;
837
838 //Send increased packet
839 int newMtu = NetConstants.PossibleMtu[_mtuIdx + 1] - NetManager.ExtraPacketSizeForLayer;
840 var p = _packetPool.GetPacket(newMtu);
841 p.Property = PacketProperty.MtuCheck;
842 FastBitConverter.GetBytes(p.RawData, 1, newMtu); //place into start
843 FastBitConverter.GetBytes(p.RawData, p.Size - 4, newMtu);//and end of packet
844
845 //Must check result for MTU fix
846 if (NetManager.SendRawAndRecycle(p, EndPoint) <= 0)
847 _finishMtu = true;
848 }

851 internal ConnectRequestResult ProcessConnectRequest(NetConnectRequestPacket connRequest)
852 {
853 //current or new request
854 switch (_connectionState)
855 {
856 //P2P case
857 case ConnectionState.Outgoing:
858 //fast check
859 if (connRequest.ConnectionTime < _connectTime)
860 {
861 return ConnectRequestResult.P2PLose;
862 }
863 //slow rare case check
864 else if (connRequest.ConnectionTime == _connectTime)
865 {
866 var remoteBytes = EndPoint.Serialize();
867 var localBytes = connRequest.TargetAddress;
868 for (int i = remoteBytes.Size-1; i >= 0; i--)
869 {
870 byte rb = remoteBytes[i];
871 if (rb == localBytes[i])
872 continue;
873 if (rb < localBytes[i])
874 return ConnectRequestResult.P2PLose;
875 }
876 }
877 break;
878
879 case ConnectionState.Connected:
880 //Old connect request
881 if (connRequest.ConnectionTime == _connectTime)
882 {
883 //just reply accept
884 NetManager.SendRaw(_connectAcceptPacket, EndPoint);
885 }
886 //New connect request
887 else if (connRequest.ConnectionTime > _connectTime)
888 {
889 return ConnectRequestResult.Reconnection;
890 }
891 break;
892
893 case ConnectionState.Disconnected:
894 case ConnectionState.ShutdownRequested:
895 if (connRequest.ConnectionTime >= _connectTime)
896 return ConnectRequestResult.NewConnection;
897 break;
898 }
899 return ConnectRequestResult.None;
900 }

902 //Process incoming packet
903 internal void ProcessPacket(NetPacket packet)
904 {
905 //not initialized
906 if (_connectionState == ConnectionState.Outgoing || _connectionState == ConnectionState.Disconnected)
907 {
908 _packetPool.Recycle(packet);
909 return;
910 }
911 if (packet.Property == PacketProperty.ShutdownOk)
912 {
913 if (_connectionState == ConnectionState.ShutdownRequested)
914 _connectionState = ConnectionState.Disconnected;
915 _packetPool.Recycle(packet);
916 return;
917 }
918 if (packet.ConnectionNumber != _connectNum)
919 {
920 NetDebug.Write(NetLogLevel.Trace, "[RR]Old packet");
921 _packetPool.Recycle(packet);
922 return;
923 }
924 Interlocked.Exchange(ref _timeSinceLastPacket, 0);
925
926 NetDebug.Write("[RR]PacketProperty: {0}", packet.Property);
927 switch (packet.Property)
928 {
929 case PacketProperty.Merged:
930 int pos = NetConstants.HeaderSize;
931 while (pos < packet.Size)
932 {
933 ushort size = BitConverter.ToUInt16(packet.RawData, pos);
934 pos += 2;
935 if (packet.RawData.Length - pos < size)
936 break;
937
938 NetPacket mergedPacket = _packetPool.GetPacket(size);
939 Buffer.BlockCopy(packet.RawData, pos, mergedPacket.RawData, 0, size);
940 mergedPacket.Size = size;
941
942 if (!mergedPacket.Verify())
943 break;
944
945 pos += size;
946 ProcessPacket(mergedPacket);
947 }
948 _packetPool.Recycle(packet);
949 break;
950 //If we get ping, send pong
951 case PacketProperty.Ping:
952 if (NetUtils.RelativeSequenceNumber(packet.Sequence, _pongPacket.Sequence) > 0)

957 NetManager.SendRaw(_pongPacket, EndPoint);
958 }

971 NetDebug.Write("[PP]Ping: {0} - {1} - {2}", packet.Sequence, elapsedMs, _remoteDelta);
972 }
973 _packetPool.Recycle(packet);
974 break;
975
976 case PacketProperty.Ack:
977 case PacketProperty.Channeled:
978 if (packet.ChannelId > _channels.Length)
979 {
980 _packetPool.Recycle(packet);
981 break;
982 }

985 {

987 _packetPool.Recycle(packet);
988 }
989 break;
990
991 //Simple packet without acks
992 case PacketProperty.Unreliable:
993 NetManager.CreateReceiveEvent(packet, DeliveryMethod.Unreliable, NetConstants.HeaderSize, this);
994 return;
995
996 case PacketProperty.MtuCheck:
997 case PacketProperty.MtuOk:
998 ProcessMtuPacket(packet);
999 break;

1001 default:

1061 internal void Update(int deltaTime)
1062 {
1063 Interlocked.Add(ref _timeSinceLastPacket, deltaTime);
1064 switch (_connectionState)
1065 {
1066 case ConnectionState.Connected:
1067 if (_timeSinceLastPacket > NetManager.DisconnectTimeout)
1068 {
1069 NetDebug.Write(
1070 "[UPDATE] Disconnect by timeout: {0} > {1}",
1071 _timeSinceLastPacket,
1072 NetManager.DisconnectTimeout);
1073 NetManager.DisconnectPeerForce(this, DisconnectReason.Timeout, 0, null);
1074 return;
1075 }
1076 break;
1077
1078 case ConnectionState.ShutdownRequested:
1079 if (_timeSinceLastPacket > NetManager.DisconnectTimeout)
1080 {
1081 _connectionState = ConnectionState.Disconnected;
1082 }
1083 else
1084 {
1085 _shutdownTimer += deltaTime;
1086 if (_shutdownTimer >= ShutdownDelay)
1087 {
1088 _shutdownTimer = 0;
1089 NetManager.SendRaw(_shutdownPacket, EndPoint);
1090 }
1091 }
1092 return;
1093
1094 case ConnectionState.Outgoing:
1095 _connectTimer += deltaTime;
1096 if (_connectTimer > NetManager.ReconnectDelay)
1097 {
1098 _connectTimer = 0;
1099 _connectAttempts++;
1100 if (_connectAttempts > NetManager.MaxConnectAttempts)
1101 {
1102 NetManager.DisconnectPeerForce(this, DisconnectReason.ConnectionFailed, 0, null);
1103 return;
1104 }
1105
1106 //else send connect again
1107 NetManager.SendRaw(_connectRequestPacket, EndPoint);
1108 }
1109 return;
1110
1111 case ConnectionState.Disconnected:
1112 return;
1113 }
1114
1115 //Send ping
1116 _pingSendTimer += deltaTime;
1117 if (_pingSendTimer >= NetManager.PingInterval)
1118 {
1119 NetDebug.Write("[PP] Send ping...");
1120 //reset timer
1121 _pingSendTimer = 0;
1122 //send ping
1123 _pingPacket.Sequence++;
1124 //ping timeout
1125 if (_pingTimer.IsRunning)
1126 UpdateRoundTripTime((int)_pingTimer.ElapsedMilliseconds);
1127 _pingTimer.Reset();
1128 _pingTimer.Start();
1129 NetManager.SendRaw(_pingPacket, EndPoint);
1130 }
1131
1132 //RTT - round trip time
1133 _rttResetTimer += deltaTime;
1134 if (_rttResetTimer >= NetManager.PingInterval * 3)
1135 {
1136 _rttResetTimer = 0;
1137 _rtt = _avgRtt;
1138 _rttCount = 1;
1139 }
1140
1141 UpdateMtuLogic(deltaTime);
1142
1143 //Pending send
1144 if (_channelSendQueue.Count > 0)
1145 {
1146 lock (_channelSendQueue)
1147 {
1148 var count = _channelSendQueue.Count;
1149 while (count-- > 0)
1150 {
1151 BaseChannel channel = _channelSendQueue.Dequeue();
1152 if (channel.SendAndCheckQueue())
1153 {
1154 // still has something to send, re-add it to the send queue
1155 _channelSendQueue.Enqueue(channel);
1156 }
1157 }
1158 }
1159 }
1160
1161 if (_unreliableChannel.Count > 0)
1162 {
1163 lock (_unreliableChannel)
1164 {
1165 while (_unreliableChannel.Count > 0)
1166 {
1167 NetPacket packet = _unreliableChannel.Dequeue();
1168 SendUserData(packet);
1169 NetManager.NetPacketPool.Recycle(packet);
1170 }
1171 }
1172 }
1173
1174 SendMerged();
1175 }
