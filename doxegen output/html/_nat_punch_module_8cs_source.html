131 //First packet (server) send to client

142 public void PollEvents()
143 {
144 if (_natPunchListener == null || (_successEvents.Count == 0 && _requestEvents.Count == 0))
145 return;
146 lock (_successEvents)
147 {
148 while (_successEvents.Count > 0)
149 {
150 var evt = _successEvents.Dequeue();
151 _natPunchListener.OnNatIntroductionSuccess(
152 evt.TargetEndPoint,
153 evt.Type,
154 evt.Token);
155 }
156 }
157 lock (_requestEvents)
158 {
159 while (_requestEvents.Count > 0)
160 {
161 var evt = _requestEvents.Dequeue();
162 _natPunchListener.OnNatIntroductionRequest(evt.LocalEndPoint, evt.RemoteEndPoint, evt.Token);
163 }
164 }
165 }

204 //We got introduce and must punch
205 private void OnNatIntroductionResponse(NatIntroduceResponsePacket req)
206 {
207 NetDebug.Write(NetLogLevel.Trace, "[NAT] introduction received");
208
209 // send internal punch
210 var punchPacket = new NatPunchPacket {Token = req.Token};
211 Send(punchPacket, req.Internal);
212 NetDebug.Write(NetLogLevel.Trace, "[NAT] internal punch sent to " + req.Internal);
213
214 // hack for some routers
215 SocketError errorCode = 0;
216 _socket.Ttl = 2;
217 _socket.SendTo(new[] { (byte)PacketProperty.Empty }, 0, 1, req.External, ref errorCode);
218
219 // send external punch
220 _socket.Ttl = NetConstants.SocketTTL;
221 punchPacket.IsExternal = true;
222 Send(punchPacket, req.External);
223 NetDebug.Write(NetLogLevel.Trace, "[NAT] external punch sent to " + req.External);
224 }
225
226 //We got punch and can connect
227 private void OnNatPunch(NatPunchPacket req, IPEndPoint senderEndPoint)
228 {
229 //Read info
230 NetDebug.Write(NetLogLevel.Trace, "[NAT] punch received from {0} - additional info: {1}",
231 senderEndPoint, req.Token);
232
233 //Release punch success to client; enabling him to Connect() to Sender if token is ok
234 lock (_successEvents)
235 {
236 _successEvents.Enqueue(new SuccessEventData
237 {
238 TargetEndPoint = senderEndPoint,
239 Type = req.IsExternal ? NatAddressType.External : NatAddressType.Internal,
240 Token = req.Token
241 });
242 }
243 }
