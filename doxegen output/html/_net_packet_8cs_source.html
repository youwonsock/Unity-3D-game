185 public static int GetProtocolId(NetPacket packet)
186 {
187 return BitConverter.ToInt32(packet.RawData, 1);

190 public static NetConnectRequestPacket FromData(NetPacket packet)
191 {
192 if (packet.ConnectionNumber >= NetConstants.MaxConnectionNumber)
193 return null;
194
195 //Getting new id for peer
196 long connectionId = BitConverter.ToInt64(packet.RawData, 5);
197
198 //Get target address
199 int addrSize = packet.RawData[13];
200 if (addrSize != 16 && addrSize != 28)
201 return null;
202 byte[] addressBytes = new byte[addrSize];
203 Buffer.BlockCopy(packet.RawData, 14, addressBytes, 0, addrSize);
204
205 // Read data and create request
206 var reader = new NetDataReader(null, 0, 0);
207 if (packet.Size > HeaderSize+addrSize)
208 reader.SetSource(packet.RawData, HeaderSize + addrSize, packet.Size);
209
210 return new NetConnectRequestPacket(connectionId, packet.ConnectionNumber, addressBytes, reader);
211 }
212
213 public static NetPacket Make(NetDataWriter connectData, SocketAddress addressBytes, long connectId)
214 {
215 //Make initial packet
216 var packet = new NetPacket(PacketProperty.ConnectRequest, connectData.Length+addressBytes.Size);
217
218 //Add data
219 FastBitConverter.GetBytes(packet.RawData, 1, NetConstants.ProtocolId);
220 FastBitConverter.GetBytes(packet.RawData, 5, connectId);
221 packet.RawData[13] = (byte)addressBytes.Size;
222 for (int i = 0; i < addressBytes.Size; i++)
223 packet.RawData[14+i] = addressBytes[i];
224 Buffer.BlockCopy(connectData.Data, 0, packet.RawData, 14+addressBytes.Size, connectData.Length);
225 return packet;
226 }
