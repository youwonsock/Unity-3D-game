479 internal int SendRaw(byte[] message, int start, int length, IPEndPoint remoteEndPoint)
480 {
481 if (!_socket.IsRunning)
482 return 0;
483
484 SocketError errorCode = 0;
485 int result;
486 if (_extraPacketLayer != null)
487 {
488 var expandedPacket = NetPacketPool.GetPacket(length + _extraPacketLayer.ExtraPacketSizeForLayer);
489 Buffer.BlockCopy(message, start, expandedPacket.RawData, 0, length);
490 int newStart = 0;
491 _extraPacketLayer.ProcessOutBoundPacket(remoteEndPoint, ref expandedPacket.RawData, ref newStart, ref length);
492 result = _socket.SendTo(expandedPacket.RawData, newStart, length, remoteEndPoint, ref errorCode);
493 NetPacketPool.Recycle(expandedPacket);
494 }
495 else
496 {
497 result = _socket.SendTo(message, start, length, remoteEndPoint, ref errorCode);
498 }
499
500 switch (errorCode)
501 {
502 case SocketError.MessageSize:
503 NetDebug.Write(NetLogLevel.Trace, "[SRD] 10040, datalen: {0}", length);
504 return -1;
505
506 case SocketError.HostUnreachable:
507 case SocketError.NetworkUnreachable:
508 NetPeer fromPeer;
509 if (DisconnectOnUnreachable && TryGetPeer(remoteEndPoint, out fromPeer))
510 {
511 DisconnectPeerForce(
512 fromPeer,
513 errorCode == SocketError.HostUnreachable ? DisconnectReason.HostUnreachable : DisconnectReason.NetworkUnreachable,
514 errorCode,
515 null);
516 }
517 CreateEvent(NetEvent.EType.Error, remoteEndPoint: remoteEndPoint, errorCode: errorCode);
518 return -1;
519 }
520 if (result <= 0)
521 return 0;
522
523 if (EnableStatistics)
524 {
525 Statistics.IncrementPacketsSent();
526 Statistics.AddBytesSent(length);
527 }
528
529 return result;
530 }

564 private void CreateEvent(
565 NetEvent.EType type,
566 NetPeer peer = null,
567 IPEndPoint remoteEndPoint = null,
568 SocketError errorCode = 0,
569 int latency = 0,
570 DisconnectReason disconnectReason = DisconnectReason.ConnectionFailed,
571 ConnectionRequest connectionRequest = null,
572 DeliveryMethod deliveryMethod = DeliveryMethod.Unreliable,
573 NetPacket readerSource = null,
574 object userData = null)
575 {
576 NetEvent evt;
577 bool unsyncEvent = UnsyncedEvents;
578
579 if (type == NetEvent.EType.Connect)
580 Interlocked.Increment(ref _connectedPeersCount);
581 else if (type == NetEvent.EType.MessageDelivered)
582 unsyncEvent = UnsyncedDeliveryEvent;
583
584 lock(_eventLock)
585 {
586 evt = _netEventPoolHead;
587 if (evt == null)
588 evt = new NetEvent(this);
589 else
590 _netEventPoolHead = evt.Next;
591 }
592
593 evt.Type = type;
594 evt.DataReader.SetSource(readerSource, readerSource == null ? 0 : readerSource.GetHeaderSize());
595 evt.Peer = peer;
596 evt.RemoteEndPoint = remoteEndPoint;
597 evt.Latency = latency;
598 evt.ErrorCode = errorCode;
599 evt.DisconnectReason = disconnectReason;
600 evt.ConnectionRequest = connectionRequest;
601 evt.DeliveryMethod = deliveryMethod;
602 evt.UserData = userData;
603
604 if (unsyncEvent || _manualMode)
605 {
606 ProcessEvent(evt);
607 }
608 else
609 {
610 lock (_netEventsQueue)
611 _netEventsQueue.Enqueue(evt);
612 }

615 private void ProcessEvent(NetEvent evt)
616 {
617 NetDebug.Write("[NM] Processing event: " + evt.Type);
618 bool emptyData = evt.DataReader.IsNull;
619 switch (evt.Type)
620 {
621 case NetEvent.EType.Connect:
622 _netEventListener.OnPeerConnected(evt.Peer);
623 break;
624 case NetEvent.EType.Disconnect:
625 var info = new DisconnectInfo
626 {
627 Reason = evt.DisconnectReason,
628 AdditionalData = evt.DataReader,
629 SocketErrorCode = evt.ErrorCode
630 };
631 _netEventListener.OnPeerDisconnected(evt.Peer, info);
632 break;
633 case NetEvent.EType.Receive:
634 _netEventListener.OnNetworkReceive(evt.Peer, evt.DataReader, evt.DeliveryMethod);
635 break;
636 case NetEvent.EType.ReceiveUnconnected:
637 _netEventListener.OnNetworkReceiveUnconnected(evt.RemoteEndPoint, evt.DataReader, UnconnectedMessageType.BasicMessage);
638 break;
639 case NetEvent.EType.Broadcast:
640 _netEventListener.OnNetworkReceiveUnconnected(evt.RemoteEndPoint, evt.DataReader, UnconnectedMessageType.Broadcast);
641 break;
642 case NetEvent.EType.Error:
643 _netEventListener.OnNetworkError(evt.RemoteEndPoint, evt.ErrorCode);
644 break;
645 case NetEvent.EType.ConnectionLatencyUpdated:
646 _netEventListener.OnNetworkLatencyUpdate(evt.Peer, evt.Latency);
647 break;
648 case NetEvent.EType.ConnectionRequest:
649 _netEventListener.OnConnectionRequest(evt.ConnectionRequest);
650 break;
651 case NetEvent.EType.MessageDelivered:
652 _deliveryEventListener.OnMessageDelivered(evt.Peer, evt.UserData);
653 break;
654 }
655 //Recycle if not message
656 if (emptyData)
657 RecycleEvent(evt);
658 else if (AutoRecycle)
659 evt.DataReader.RecycleInternal();
660 }

675 //Update function
676 private void UpdateLogic()
677 {
678 var peersToRemove = new List<NetPeer>();
679 var stopwatch = new Stopwatch();
680 stopwatch.Start();
681
682 while (_socket.IsRunning)
683 {
684 ProcessDelayedPackets();
685 int elapsed = (int)stopwatch.ElapsedMilliseconds;
686 elapsed = elapsed <= 0 ? 1 : elapsed;
687 stopwatch.Reset();
688 stopwatch.Start();
689
690 for (var netPeer = _headPeer; netPeer != null; netPeer = netPeer.NextPeer)
691 {
692 if (netPeer.ConnectionState == ConnectionState.Disconnected && netPeer.TimeSinceLastPacket > DisconnectTimeout)
693 {
694 peersToRemove.Add(netPeer);
695 }
696 else
697 {
698 netPeer.Update(elapsed);
699 }
700 }
701 if (peersToRemove.Count > 0)
702 {
703 _peersLock.EnterWriteLock();
704 for (int i = 0; i < peersToRemove.Count; i++)
705 RemovePeerInternal(peersToRemove[i]);
706 _peersLock.ExitWriteLock();
707 peersToRemove.Clear();
708 }
709 ProcessNtpRequests(elapsed);
710
711 int sleepTime = UpdateTime - (int)stopwatch.ElapsedMilliseconds;
712 if (sleepTime > 0)
713 _updateTriggerEvent.WaitOne(sleepTime);
714 }
715 stopwatch.Stop();
716 }

846 internal NetPeer OnConnectionSolved(ConnectionRequest request, byte[] rejectData, int start, int length)
847 {
848 NetPeer netPeer = null;
849
850 if (request.Result == ConnectionRequestResult.RejectForce)
851 {
852 NetDebug.Write(NetLogLevel.Trace, "[NM] Peer connect reject force.");
853 if (rejectData != null && length > 0)
854 {
855 var shutdownPacket = NetPacketPool.GetWithProperty(PacketProperty.Disconnect, length);
856 shutdownPacket.ConnectionNumber = request.ConnectionNumber;
857 FastBitConverter.GetBytes(shutdownPacket.RawData, 1, request.ConnectionTime);
858 if (shutdownPacket.Size >= NetConstants.PossibleMtu[0])
859 NetDebug.WriteError("[Peer] Disconnect additional data size more than MTU!");
860 else
861 Buffer.BlockCopy(rejectData, start, shutdownPacket.RawData, 9, length);
862 SendRawAndRecycle(shutdownPacket, request.RemoteEndPoint);
863 }
864 }
865 else
866 {
867 _peersLock.EnterUpgradeableReadLock();
868 if (_peersDict.TryGetValue(request.RemoteEndPoint, out netPeer))
869 {
870 //already have peer
871 _peersLock.ExitUpgradeableReadLock();
872 }
873 else if (request.Result == ConnectionRequestResult.Reject)
874 {
875 netPeer = new NetPeer(this, request.RemoteEndPoint, GetNextPeerId());
876 netPeer.Reject(request.ConnectionTime, request.ConnectionNumber, rejectData, start, length);
877 AddPeer(netPeer);
878 _peersLock.ExitUpgradeableReadLock();
879 NetDebug.Write(NetLogLevel.Trace, "[NM] Peer connect reject.");
880 }
881 else //Accept
882 {
883 netPeer = new NetPeer(this, request.RemoteEndPoint, GetNextPeerId(), request.ConnectionTime, request.ConnectionNumber);
884 AddPeer(netPeer);
885 _peersLock.ExitUpgradeableReadLock();
886 CreateEvent(NetEvent.EType.Connect, netPeer);
887 NetDebug.Write(NetLogLevel.Trace, "[NM] Received peer connection Id: {0}, EP: {1}",
888 netPeer.ConnectTime, netPeer.EndPoint);
889 }
890 }
891
892 lock(_requestsDict)
893 _requestsDict.Remove(request.RemoteEndPoint);
894
895 return netPeer;
896 }

904 private void ProcessConnectRequest(
905 IPEndPoint remoteEndPoint,
906 NetPeer netPeer,
907 NetConnectRequestPacket connRequest)
908 {
909 byte connectionNumber = connRequest.ConnectionNumber;
910 ConnectionRequest req;
911
912 //if we have peer
913 if (netPeer != null)
914 {
915 var processResult = netPeer.ProcessConnectRequest(connRequest);
916 NetDebug.Write("ConnectRequest LastId: {0}, NewId: {1}, EP: {2}, Result: {3}",
917 netPeer.ConnectTime,
918 connRequest.ConnectionTime,
919 remoteEndPoint,
920 processResult);
921
922 switch (processResult)
923 {
924 case ConnectRequestResult.Reconnection:
925 DisconnectPeerForce(netPeer, DisconnectReason.Reconnect, 0, null);
926 RemovePeer(netPeer);
927 //go to new connection
928 break;
929 case ConnectRequestResult.NewConnection:
930 RemovePeer(netPeer);
931 //go to new connection
932 break;
933 case ConnectRequestResult.P2PLose:
934 DisconnectPeerForce(netPeer, DisconnectReason.PeerToPeerConnection, 0, null);
935 RemovePeer(netPeer);
936 //go to new connection
937 break;
938 default:
939 //no operations needed
940 return;
941 }
942 //ConnectRequestResult.NewConnection
943 //Set next connection number
944 if(processResult != ConnectRequestResult.P2PLose)
945 connectionNumber = (byte)((netPeer.ConnectionNum + 1) % NetConstants.MaxConnectionNumber);
946 //To reconnect peer
947 }
948 else
949 {
950 NetDebug.Write("ConnectRequest Id: {0}, EP: {1}", connRequest.ConnectionTime, remoteEndPoint);
951 }
952
953 lock (_requestsDict)
954 {
955 if (_requestsDict.TryGetValue(remoteEndPoint, out req))
956 {
957 req.UpdateRequest(connRequest);
958 return;
959 }
960 req = new ConnectionRequest(
961 connRequest.ConnectionTime,
962 connectionNumber,
963 connRequest.Data,
964 remoteEndPoint,
965 this);
966 _requestsDict.Add(remoteEndPoint, req);
967 }
968 NetDebug.Write("[NM] Creating request event: " + connRequest.ConnectionTime);
969 CreateEvent(NetEvent.EType.ConnectionRequest, connectionRequest: req);

972 private void DataReceived(NetPacket packet, IPEndPoint remoteEndPoint)
973 {
974 if (EnableStatistics)
975 {
976 Statistics.IncrementPacketsReceived();
977 Statistics.AddBytesReceived(packet.Size);
978 }
979
980 if (_ntpRequests.Count > 0)
981 {
982 NtpRequest request;
983 if (_ntpRequests.TryGetValue(remoteEndPoint, out request))
984 {
985 if (packet.Size < 48)
986 {
987 NetDebug.Write(NetLogLevel.Trace, "NTP response too short: {}", packet.Size);
988 return;
989 }
990
991 byte[] copiedData = new byte[packet.Size];
992 Buffer.BlockCopy(packet.RawData, 0, copiedData, 0, packet.Size);
993 NtpPacket ntpPacket = NtpPacket.FromServerResponse(copiedData, DateTime.UtcNow);
994 try
995 {
996 ntpPacket.ValidateReply();
997 }
998 catch (InvalidOperationException ex)
999 {
1000 NetDebug.Write(NetLogLevel.Trace, "NTP response error: {}", ex.Message);
1001 ntpPacket = null;
1002 }
1003
1004 if (ntpPacket != null)
1005 {
1006 _ntpRequests.Remove(remoteEndPoint);
1007 if(_ntpEventListener != null)
1008 _ntpEventListener.OnNtpResponse(ntpPacket);
1009 }
1010 return;
1011 }
1012 }
1013
1014 if (_extraPacketLayer != null)
1015 {
1016 int start = 0;
1017 _extraPacketLayer.ProcessInboundPacket(remoteEndPoint, ref packet.RawData, ref start, ref packet.Size);
1018 if (packet.Size == 0)
1019 return;
1020 }
1021
1022 if (!packet.Verify())
1023 {
1024 NetDebug.WriteError("[NM] DataReceived: bad!");
1025 NetPacketPool.Recycle(packet);
1026 return;
1027 }
1028
1029 switch (packet.Property)
1030 {
1031 //special case connect request
1032 case PacketProperty.ConnectRequest:
1033 if (NetConnectRequestPacket.GetProtocolId(packet) != NetConstants.ProtocolId)
1034 {
1035 SendRawAndRecycle(NetPacketPool.GetWithProperty(PacketProperty.InvalidProtocol), remoteEndPoint);
1036 return;
1037 }
1038 break;
1039 //unconnected messages
1040 case PacketProperty.Broadcast:
1041 if (!BroadcastReceiveEnabled)
1042 return;
1043 CreateEvent(NetEvent.EType.Broadcast, remoteEndPoint: remoteEndPoint, readerSource: packet);
1044 return;
1045 case PacketProperty.UnconnectedMessage:
1046 if (!UnconnectedMessagesEnabled)
1047 return;
1048 CreateEvent(NetEvent.EType.ReceiveUnconnected, remoteEndPoint: remoteEndPoint, readerSource: packet);
1049 return;
1050 case PacketProperty.NatMessage:
1051 if (NatPunchEnabled)
1052 NatPunchModule.ProcessMessage(remoteEndPoint, packet);
1053 return;
1054 }
1055
1056 //Check normal packets
1057 NetPeer netPeer;
1058 _peersLock.EnterReadLock();
1059 bool peerFound = _peersDict.TryGetValue(remoteEndPoint, out netPeer);
1060 _peersLock.ExitReadLock();
1061
1062 switch (packet.Property)
1063 {
1064 case PacketProperty.ConnectRequest:
1065 var connRequest = NetConnectRequestPacket.FromData(packet);
1066 if (connRequest != null)
1067 ProcessConnectRequest(remoteEndPoint, netPeer, connRequest);
1068 break;
1069 case PacketProperty.PeerNotFound:
1070 if (peerFound)
1071 {
1072 if (netPeer.ConnectionState != ConnectionState.Connected)
1073 return;
1074 if (packet.Size == 1)
1075 {
1076 //first reply
1077 var p = NetPacketPool.GetWithProperty(PacketProperty.PeerNotFound, 9);
1078 p.RawData[1] = 0;
1079 FastBitConverter.GetBytes(p.RawData, 2, netPeer.ConnectTime);
1080 SendRawAndRecycle(p, remoteEndPoint);
1081 NetDebug.Write("PeerNotFound sending connectTime: {0}", netPeer.ConnectTime);
1082 }
1083 else if (packet.Size == 10 && packet.RawData[1] == 1 && BitConverter.ToInt64(packet.RawData, 2) == netPeer.ConnectTime)
1084 {
1085 //second reply
1086 NetDebug.Write("PeerNotFound received our connectTime: {0}", netPeer.ConnectTime);
1087 DisconnectPeerForce(netPeer, DisconnectReason.RemoteConnectionClose, 0, null);
1088 }
1089 }
1090 else if (packet.Size == 10 && packet.RawData[1] == 0)
1091 {
1092 //send reply back
1093 packet.RawData[1] = 1;
1094 SendRawAndRecycle(packet, remoteEndPoint);
1095 }
1096 break;
1097 case PacketProperty.InvalidProtocol:
1098 if (peerFound && netPeer.ConnectionState == ConnectionState.Outgoing)
1099 DisconnectPeerForce(netPeer, DisconnectReason.InvalidProtocol, 0, null);
1100 break;
1101 case PacketProperty.Disconnect:
1102 if (peerFound)
1103 {
1104 var disconnectResult = netPeer.ProcessDisconnect(packet);
1105 if (disconnectResult == DisconnectResult.None)
1106 {
1107 NetPacketPool.Recycle(packet);
1108 return;
1109 }
1110 DisconnectPeerForce(
1111 netPeer,
1112 disconnectResult == DisconnectResult.Disconnect
1113 ? DisconnectReason.RemoteConnectionClose
1114 : DisconnectReason.ConnectionRejected,
1115 0, packet);
1116 }
1117 else
1118 {
1119 NetPacketPool.Recycle(packet);
1120 }
1121 //Send shutdown
1122 SendRawAndRecycle(NetPacketPool.GetWithProperty(PacketProperty.ShutdownOk), remoteEndPoint);
1123 break;
1124 case PacketProperty.ConnectAccept:
1125 if (!peerFound)
1126 return;
1127 var connAccept = NetConnectAcceptPacket.FromData(packet);
1128 if (connAccept != null && netPeer.ProcessConnectAccept(connAccept))
1129 CreateEvent(NetEvent.EType.Connect, netPeer);
1130 break;
1131 default:
1132 if(peerFound)
1133 netPeer.ProcessPacket(packet);
1134 else
1135 SendRawAndRecycle(NetPacketPool.GetWithProperty(PacketProperty.PeerNotFound), remoteEndPoint);
1136 break;
1137 }
1138 }
