79 public static void GetLocalIpList(IList<string> targetList, LocalAddrType addrType)
80 {
81 bool ipv4 = (addrType & LocalAddrType.IPv4) == LocalAddrType.IPv4;
82 bool ipv6 = (addrType & LocalAddrType.IPv6) == LocalAddrType.IPv6;
83 try
84 {
85 foreach (NetworkInterface ni in NetworkInterface.GetAllNetworkInterfaces())
86 {
87 //Skip loopback and disabled network interfaces
88 if (ni.NetworkInterfaceType == NetworkInterfaceType.Loopback ||
89 ni.OperationalStatus != OperationalStatus.Up)
90 continue;
91
92 var ipProps = ni.GetIPProperties();
93
94 //Skip address without gateway
95 if (ipProps.GatewayAddresses.Count == 0)
96 continue;
97
98 foreach (UnicastIPAddressInformation ip in ipProps.UnicastAddresses)
99 {
100 var address = ip.Address;
101 if ((ipv4 && address.AddressFamily == AddressFamily.InterNetwork) ||
102 (ipv6 && address.AddressFamily == AddressFamily.InterNetworkV6))
103 targetList.Add(address.ToString());
104 }
105 }
106 }
107 catch
108 {
109 //ignored
110 }
111
112 //Fallback mode (unity android)
113 if (targetList.Count == 0)
114 {
115 IPAddress[] addresses = Dns.GetHostEntry(Dns.GetHostName()).AddressList;
116 foreach (IPAddress ip in addresses)
117 {
118 if((ipv4 && ip.AddressFamily == AddressFamily.InterNetwork) ||
119 (ipv6 && ip.AddressFamily == AddressFamily.InterNetworkV6))
120 targetList.Add(ip.ToString());
121 }
122 }
123 if (targetList.Count == 0)
124 {
125 if(ipv4)
126 targetList.Add("127.0.0.1");
127 if(ipv6)
128 targetList.Add("::1");
129 }
130 }
