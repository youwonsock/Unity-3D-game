196 void QueueLog(string logString, string stackTrace, LogType type)
197 {
198 if(logs.Count > 1000)
199 {
200 while(logs.Count > 1000)
201 {
202 logs.RemoveAt(0);
203 }
204 }
205
206 #if CSHARP_7_3_OR_NEWER
207 logString = $"{logString}\n{stackTrace}\n";
208 logs.Add(new QueuedLog() { message = logString, logType = type.ToString(), timestamp = $"[{DateTime.Now.ToString("HH:mm:ss")}]" } );
209 #else
210 logString = logString + "\n" + stackTrace + "\n";
211 logs.Add(new QueuedLog() { message = logString, logType = type.ToString(), timestamp = "[" + DateTime.Now.ToString("HH:mm:ss") + "]" } );

215 void LateUpdate()
216 {

234 string cMessage = "";

236 foreach(var cLog in logs)

238 cMessage = JsonUtility.ToJson(cLog);
239 _dataWriter.Reset();
240 _dataWriter.Put(cMessage);
